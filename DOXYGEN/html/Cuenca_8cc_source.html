11BinTree<pair<int,int>> Cuenca::obtener_arbol_de_beneficios(int id_prod_a_comprar, int id_prod_a_vender, int productos_a_comprar, int productos_a_vender, const BinTree<string>& nodo){
12 // Caso Base ########################################################################################
13
14 // Si el nodo es vacío
15 if (nodo.empty()) return BinTree<pair<int,int>>();
16
17 // No hay más productos que vender
18 if (productos_a_comprar + productos_a_vender == 0) return BinTree<pair<int,int>>();
19
20 // Paso Inductivo ###################################################################################
21
22 // Buscamos la ciudad actual
23 this->_it_mapa_ciudad = this->_mapa_ciudades.find(nodo.value());
24
25 // Obtenemos la ciudad actual
26 Ciudad ciudad_seleccionada = this->_it_mapa_ciudad->second;
27
28 // Obtenemos las cantidades poseídas y necesarias del producto que el barco quiere comprar y vender.
29 pair<int,int> info_produ_comprar;
30 bool proceder_c = ciudad_seleccionada.consultar_prod_c(id_prod_a_comprar,info_produ_comprar, false);
31
32 pair<int,int> info_produ_vender;
33 bool proceder_v = ciudad_seleccionada.consultar_prod_c(id_prod_a_vender,info_produ_vender, false);
34 //-------------------------------------------------------------------------------------------------
35
36 // Variable de productos comprados y vendidos
37 int total_productos_comprados = 0;
38 int total_productos_vendidos = 0;
39 //-------------------------------------------
40
41 // Procedemos los cálculos:
42
43 // Cálculo de productos comprados en cada ciudad
44 if ((proceder_c) and (productos_a_comprar > 0))
45 {
46 // Obtenemos la cantidad que sobra
47 int productos_sobrantes = info_produ_comprar.first - info_produ_comprar.second;
48 // Sí sobran productos a la ciudad
49 if (productos_sobrantes > 0){
50 // Si podemos comprar todos los productos que barco quiere en una ciudad, entonces compramos todo lo que necesitamos
51 if (productos_sobrantes >= productos_a_comprar){
52 total_productos_comprados = productos_a_comprar;
53 productos_a_comprar = 0;
54 }
55 else { // De lo contrario compraremos los productos que podamos
56 total_productos_comprados = productos_sobrantes;
57 productos_a_comprar -= productos_sobrantes;
58 }
59 }
60 }
61
62 // Cálculo de productos vendidos en cada ciudad
63 if ((proceder_v) and (productos_a_vender > 0))
64 {
65 // Obtenemos la cantidad que necesita
66 int productos_neccesarios = info_produ_vender.second - info_produ_vender.first;
67 // Si podemos vender todos los productos que barco quiere en una ciudad, entonces venderemos todo lo que neccesite
68 if (productos_neccesarios > 0){
69 if (productos_neccesarios >= productos_a_vender){
70 total_productos_vendidos = productos_a_vender;
71 productos_a_vender = 0;
72 }
73 else { // De lo contrario venderemos los productos que podamos
74 total_productos_vendidos = productos_neccesarios;
75 productos_a_vender -= productos_neccesarios;
76 }
77 }
78 }
79
80 // Construimos un "pair" con las compras y ventas realizadas en la ciudad
81 pair<int,int> beneficio;
82
83 beneficio.first = total_productos_comprados;
84 beneficio.second = total_productos_vendidos;
85 //-----------------------------------------------------------------------
86
87 // Realizamos el recorrido recursivo por la izquierda y después por la derecha
88 BinTree<pair<int,int>> arbol_izqueirda = obtener_arbol_de_beneficios(id_prod_a_comprar,id_prod_a_vender,productos_a_comprar,productos_a_vender,nodo.left());
89 BinTree<pair<int,int>> arbol_derecha = obtener_arbol_de_beneficios(id_prod_a_comprar,id_prod_a_vender,productos_a_comprar,productos_a_vender,nodo.right());
90
91 // Devolvemos un árbol con el beneficio conseguido en la ciudad y con su árbol izquierdo y derecho
92 return BinTree<pair<int,int>>(beneficio,arbol_izqueirda,arbol_derecha);
93}

96list<pair<char,int>> Cuenca::recorrido_con_max_beneficio(const BinTree<pair<int,int>>& nodo){
97 // Caso Base ########################################################################################
98
99 // Si el nodo es vacío
100 if (nodo.empty()) return list<pair<char,int>>();
101
102 // Paso Inductivo ###################################################################################
103
104 // Construimos las listas postorden
105 list<pair<char,int>> recorido_izquierda = recorrido_con_max_beneficio(nodo.left());
106 list<pair<char,int>> recorido_derecha = recorrido_con_max_beneficio(nodo.right());
107 //---------------------------------
108
109 // Inicializamos los beneficios por ruta
110 int beneficio_izquierda = 0, beneficio_derecha = 0;
111
112 // Beneficios por ruta
113 if (!recorido_izquierda.empty()) beneficio_izquierda = recorido_izquierda.front().second;
114 if (!recorido_derecha.empty()) beneficio_derecha = recorido_derecha.front().second;
115 //--------------------
116
117 // Calculamos el beneficio en nuestra posición
118 int beneficio_nodo_actual = nodo.value().first + nodo.value().second;
119
120 // Distinguimos por diferentes casos:
121
122 // En caso de que el beneficio de la parte izquierda y la parte derecha sea 0, entonces finalizamos la función
123 if (beneficio_izquierda + beneficio_derecha == 0) {
124 // Devolvemos la lista formada únicamente por un elemento
125 list<pair<char,int>> lista_unica;
126 // Char u -> Representa una única ciudad en la lista
127 lista_unica.push_front(make_pair('u',beneficio_nodo_actual));
128 return lista_unica;
129 }
130
131 // Si el beneficio de la derecha es mayor que el de la izquierda
132 if (beneficio_derecha > beneficio_izquierda){
133 // Si beneficio_derecha és mayor que beneficio_izquierda quiere decir que es mayor que 0 en el peor de los casos cosa que quiere decir que existe.
134
135 // Incrementamos el beneficio
136 beneficio_nodo_actual += recorido_derecha.front().second;
137 // Añadimos el nuevo nodo y la dirección por la que debemos ir
138 recorido_derecha.push_front(make_pair('d',beneficio_nodo_actual));
139 // Devolvemos el recorrido hecho
140 return recorido_derecha;
141
142 } // Si el beneficio de la izquierda es mayor que el de la derecha
143 else if (beneficio_derecha < beneficio_izquierda){
144 // Si beneficio_izquierda és mayor que beneficio_izquierda quiere decir que es mayor que 0 en el peor de los casos cosa que quiere decir que existe.
145
146 // Incrementamos el beneficio
147 beneficio_nodo_actual += recorido_izquierda.front().second;
148 // Añadimos el nuevo nodo y la dirección por la que debemos ir
149 recorido_izquierda.push_front(make_pair('i',beneficio_nodo_actual));
150 // Devolvemos el recorrido hecho
151 return recorido_izquierda;
152 }
153 else { // Si el beneficio de la izquierda es igual que el de la derecha
154 // Miramos el recorrido más corto de los dos, si son iguales escogeremos el de la izquierda
155 // Si el recorrido de la derecha es más pequeño que el de la izquierda
156 if (recorido_derecha.size() < recorido_izquierda.size()){
157 // Entonces, como existe el recorrido por la derecha: (Explicado arriba)
158 // Incrementamos el beneficio
159 beneficio_nodo_actual += recorido_derecha.front().second; //Incrementamos el beneficio
160 // Añadimos el nuevo nodo y la dirección por la que debemos ir
161 recorido_derecha.push_front(make_pair('d',beneficio_nodo_actual));
162 // Devolvemos el recorrido hecho
163 return recorido_derecha;
164 }
165 else { // Si el recorrido de la izquierda es más pequeño o igual que el de la derecha
166 // Entonces, si existe el recorrido por la izquierda
167 // Incrementamos el beneficio
168 if (!recorido_izquierda.empty()) beneficio_nodo_actual += recorido_izquierda.front().second; //Incrementamos el beneficio
169 // Añadimos el nuevo nodo y la dirección por la que debemos ir
170 recorido_izquierda.push_front(make_pair('i',beneficio_nodo_actual));
171 // Devolvemos el recorrido hecho
172 return recorido_izquierda;
173 }
174 }

177void Cuenca::aux_viajar(Cjt_Productos &Productos,Barco& Barco, int id_prod_a_comprar, int id_prod_a_vender){
178 // Consultamos las cantidades a comprar y vender
179 int productos_a_comprar = Barco.consultar_cantidad_a_comprar();
180 int productos_a_vender = Barco.consultar_cantidad_a_vender();
181 //----------------------------------------------
182 // Obtenemos el árbol de beneficios
183 BinTree<pair<int,int>> Arbol_de_ventas = this->obtener_arbol_de_beneficios(id_prod_a_comprar, id_prod_a_vender,productos_a_comprar,productos_a_vender,this->_cuenca);
184 // A partir del árbol de beneficios obtenemos la ruta más beneficiosa
185 list<pair<char,int>> ruta = recorrido_con_max_beneficio(Arbol_de_ventas);
186 // Creamos un "iterdador" para consultar la ruta obtenida
187 list<pair<char,int>> ::const_iterator it_list = ruta.begin();
188
189 // Auxiliares que representan los productos que el barco quiere comprar y vender
190 int aux_productos_a_comprar = productos_a_comprar;
191 int aux_productos_a_vender = productos_a_vender;
192 //------------------------------------------------------------------------------
193 // Beneficio total de la ruta
194 int beneficio_total = 0;
195 // Primera ciudad a consultar
196 BinTree<string> nodo_actual = this->_cuenca;
197 // Última ciudad consultada
198 string ultima_ciudad = "#";
199
200 // Bucle dedicado a la modificación de los productos de las ciudades de la ruta
201 //|Si aún hay productos que comprar y vender en el barco| |No hemos llegado al final de la ruta| |El nodo actual existe|
202 while ((aux_productos_a_comprar + aux_productos_a_vender > 0) and (it_list != ruta.end()) and (!nodo_actual.empty()))
203 {
204 // Actualizamos la última ciudad consultada
205 ultima_ciudad = nodo_actual.value();
206
207 // Buscamos la ciudad cuyo nombre lo contiene el nodo
208 this->_it_mapa_ciudad = this->_mapa_ciudades.find(nodo_actual.value()); //Buscamos la ciudad actual
209
210 // Obtenemos la informacion necesaria del producto a comprar y vender
211 pair<int,int> info_produ_comprar;
212 bool proceder_c = this->_it_mapa_ciudad->second.consultar_prod_c(id_prod_a_comprar,info_produ_comprar, false);
213
214
215 pair<int,int> info_produ_vender;
216 bool proceder_v = this->_it_mapa_ciudad->second.consultar_prod_c(id_prod_a_vender,info_produ_vender, false);
217
218 //-------------------------------------------------------------------
219 // Si el producto a comprar existe y lo tiene la ciudad
220 if ((proceder_c) and (aux_productos_a_comprar > 0))
221 {
222 // Obtenemos la cantidad que sobra de la ciudad
223 int productos_sobrantes = info_produ_comprar.first - info_produ_comprar.second;
224 // Si a la ciudad le sobra el producto
225 if (productos_sobrantes > 0){
226 // Si la cantidad que sobra es mayor a la que podemos comprar compraremos todo lo que podamos
227 if (productos_sobrantes >= aux_productos_a_comprar){
228 // Modificamos las cantidades del inventario de la ciudad
229 this->_it_mapa_ciudad->second.modificar_prod_c(Productos,id_prod_a_comprar,info_produ_comprar.first - aux_productos_a_comprar,info_produ_comprar.second, false);
230 // Incrementamos el beneficio total
231 beneficio_total += aux_productos_a_comprar;
232 // Como hemos comprado todo lo que necesitamos ya no compraremos más
233 aux_productos_a_comprar = 0;
234 }
235 else { // Si la cantidad que sobra es menor a la que podemos comprar compraremos todo el producto que sobre
236 // Incrementamos el beneficio total
237 beneficio_total += productos_sobrantes;
238 // Modificamos las cantidades del inventario de la ciudad
239 this->_it_mapa_ciudad->second.modificar_prod_c(Productos,id_prod_a_comprar,info_produ_comprar.first - productos_sobrantes,info_produ_comprar.second, false);
240 // Decrementamos la cantidad que hemos comprado
241 aux_productos_a_comprar -= productos_sobrantes;
242 }
243 }
244 }
245
246 // Cálculo de productos vendidos en cada ciudad
247 if (proceder_v and (aux_productos_a_vender > 0)){
248 // Obtenemos la cantidad que necesita
249 int productos_neccesarios = info_produ_vender.second - info_produ_vender.first;
250 // Si a la ciudad necesita el producto
251 if (productos_neccesarios > 0){
252 // Si la cantidad que necesita es mayor a la que podemos vender venderemos todo lo que podamos
253 if (productos_neccesarios >= aux_productos_a_vender){
254 // Modificamos las cantidades del inventario de la ciudad
255 this->_it_mapa_ciudad->second.modificar_prod_c(Productos,id_prod_a_vender,info_produ_vender.first + aux_productos_a_vender,info_produ_vender.second, false);
256 // Incrementamos el beneficio total
257 beneficio_total += aux_productos_a_vender;
258 // Como hemos vendido todo lo que queríamos ya no venderemos más
259 aux_productos_a_vender = 0;
260 }
261 else { // Si la cantidad que necesita es menor a la que podemos vender venderemos todo el producto que necesita
262 // Modificamos las cantidades del inventario de la ciudad
263 this->_it_mapa_ciudad->second.modificar_prod_c(Productos,id_prod_a_vender,info_produ_vender.first + productos_neccesarios,info_produ_vender.second, false);
264 // Incrementamos el beneficio total
265 beneficio_total += productos_neccesarios;
266 // Decrementamos la cantidad que hemos vendido
267 aux_productos_a_vender -= productos_neccesarios;
268 }
269 }
270 }
271
272 // Seguimos por la ruta, si el char es 'i' iremos por la izquierda, de lo contrario por la derecha.
273 // Si no es ni 'i' ni 'd' entonces es 'u' y esto significa que es una lista de única ciudad, es decir,
274 // finalizamos la ruta.
275
276 if (it_list->first == 'i'){
277 nodo_actual = nodo_actual.left();
278 // Incrementamos el iterador
279 it_list++;
280 }
281 else if (it_list->first == 'd'){
282 nodo_actual = nodo_actual.right();
283 // Incrementamos el iterador
284 it_list++;
285 }
286 else if (it_list->first == 'u'){
287 it_list = ruta.end();
288 }
289 }
290 // Imprimimos el beneficio total
291 cout << beneficio_total <<endl;
292 // Si el beneficio es 0 quiere decir que el barco no modifica a ninguna ciudad y no tendremos que
293 // añadir ninguna ciudad al registro del barco
294 if (beneficio_total > 0) Barco.agregar_ultima_ciudad_consultada(ultima_ciudad);
295
296}

299void Cuenca::aux_redistribuir(Cjt_Productos& Productos, const BinTree<string>& arbol_cuenca){
300 // Si el nodo actual es vacío -> Finalizamos
301 if (arbol_cuenca.empty()) return;
302 // Si el árbol izquierdo existe, entonces podemos comerciar con este
303 if (!arbol_cuenca.left().empty()){
304 // Comerciamos con el árbol izquierdo
305 comerciar(Productos,arbol_cuenca.value(),arbol_cuenca.left().value());
306 // Redistribuimos desde el árbol izquierdo a sus hijos
307 aux_redistribuir(Productos,arbol_cuenca.left());
308 }
309 // Si el árbol derecho existe, entonces podemos comerciar con este
310 if (!arbol_cuenca.right().empty()){
311 // Comerciamos con el árbol derecho
312 comerciar(Productos,arbol_cuenca.value(),arbol_cuenca.right().value());
313 // Redistribuimos desde el árbol derecho a sus hijos
314 aux_redistribuir(Productos,arbol_cuenca.right());
315 }

502void Cuenca::quitar_prod(Cjt_Productos& Productos, string id_ciudad, int id_producto){
503 // Si el producto no existe
504 if ((id_producto <= 0) or (id_producto >= Productos.consultar_num() + 1)) {
505 // Imprimimos el error de producto no existene
506 error_notification(1);
507 //Informacion no valida con errores
508 return;
509 }
510 // Buscamos la ciudad deseada
511 this->_it_mapa_ciudad = this->_mapa_ciudades.find(id_ciudad);
512
513 // Si la ciudad no existe
514 if (_it_mapa_ciudad == this->_mapa_ciudades.end()) {
515 // Imprimimos el error de ciudad no existente y finalizamos
516 error_notification(2);
517 return;
518 }
519
520 // Si la ciudad existe
521 // Intentamos quitar el producto
522 bool proceder = _it_mapa_ciudad->second.quitar_prod_c(Productos,id_producto);
